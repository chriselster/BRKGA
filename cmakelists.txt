cmake_minimum_required(VERSION 3.12)
project(MyProject)

# Set the source files
file(GLOB SOURCES src/Entities/*.cpp)

# Set the header files
file(GLOB HEADERS src/Headers/*.hpp)

# add headers inside Headers/brkga_mp_ipr
file(GLOB HEADERS_BRKGA_MP_IPR src/Headers/brkga_mp_ipr/*.hpp)

# Add an executable target
add_executable(MyProject ${SOURCES})

# Include the header files
target_include_directories(MyProject PRIVATE src/Headers)
